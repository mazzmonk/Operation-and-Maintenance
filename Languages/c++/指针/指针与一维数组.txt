#include <stdio.h>

int main(int argc, char* argv[]) {
  int arr[] = {100, 101, 102};
  int *ptr = arr;
  int i, size = 0;
  size = (sizeof arr/sizeof (arr[0]));

  printf("使用arr指针常量来表示:\n");
  for (i = 0; i < size; i++)
    printf("&arr[%d] = %x\n", i, &arr[i]);

  printf("\n");
  for (i = 0; i < size; i++)
    printf("arr+%d = %x\n",i, arr + i);

  printf("\n");
  for(i = 0; i < size; i++)
    printf("arr[%d] = %d\n", i, arr[i]);

  printf("\n");
  for(i = 0; i < size; i++)
    printf("*(arr+%d) = %d\n", i, *(arr+i));



  printf("\n使用ptr指针变量来表示:\n");
  for (i = 0; i < size; i++)
    printf("ptr + %d = %x\n", i, ptr+i);

  printf("\n");
  for (i = 0; i < size; i++)
    printf("ptr[%d] = %d\n", i, ptr[i]);

  printf("\n");
  for (i = 0; i < size; i++)
    printf("*(ptr+%d) = %d\n", i, *(ptr+i));

  return 0;
}

使用arr指针常量来表示:
