#cpu测试
# --cpu-max-prime=N(素数发生器的上限)     threads(线程）

sysbench --cpu-max-prime=10000 --threads=8 cpu run

#硬盘I/O测试
#创建测试数据 28G
sysbench --file-total-size=10G fileio prepare
#测试数据 
#--file-test-mode=rndrw（随机读取）
#--time 测试时间
#seqwr：顺序写入
#seqrewr：顺序重写
#seqrd：顺序读取
#rndrd：随机读取
#rndwr：随机写入
#rndrw：随机读取/写入
sysbench --file-total-size=10G --file-test-mode=rndrw --time=300 --max-requests=0 fileio run
#清除测试数据
sysbench --file-total-size=10G fileio cleanup

#测试 memory
#缓冲区大小为 8KB，在内存中传输 4GB 数据的测试
sysbench --threads=16 --memory-block-size=8k --memory-total-size=4G memory run
